CREATE TABLE kartya
(
    kid INT PRIMARY KEY AUTO_INCREMENT,
    kazonosito int,
    kallapot tinyint
);

CREATE TABLE ember
(
    eid INT PRIMARY KEY AUTO_INCREMENT,
    enev varchar(50),
    email varchar(50),
    etel varchar(12),
    estatuszdolg tinyint,
    ekomment varchar(100),
    jogosultsag varchar(100)
);

CREATE TABLE jarmu
(
    jid INT PRIMARY KEY AUTO_INCREMENT,
    jrendszam varchar(8),
    jtipus varchar(20),
    jszin varchar(20),
    jkomment varchar(100)
);


CREATE TABLE kartyaember
(
    keid INT PRIMARY KEY AUTO_INCREMENT,
    kid int,
    eid int
);

CREATE TABLE emberjarmu
(
    ejid INT PRIMARY KEY AUTO_INCREMENT,
    eid int,
    jid int
);

CREATE TABLE naplo
(
    nid INT PRIMARY KEY AUTO_INCREMENT,
    keid int,
    ejid int,
    nstatusz tinyint,
    ndatumtol date,
    nidotol datetime,
    ndatumig date,
    nidoig datetime,
    nparkhely INT
);

CREATE TABLE ceg
(
    cid INT PRIMARY KEY AUTO_INCREMENT,
    cnev varchar(100)
);

CREATE TABLE ertekeles
(
    eid int PRIMARY KEY,
    megjegyzes varchar(255),
    csillag int
);
